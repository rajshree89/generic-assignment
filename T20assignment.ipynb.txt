    "def get_scorecard(innings):\n",
    "    result = []\n",
    "\n",
    "    for inn in innings:\n",
    "        deliveries = list(inn.values())[0][\"deliveries\"]\n",
    "        bat, bowl = {}, {}\n",
    "\n",
    "        for ball in deliveries:\n",
    "            info = list(ball.values())[0]\n",
    "            bat[info[\"batsman\"]] = bat.get(info[\"batsman\"], 0) + info[\"runs\"][\"batsman\"]\n",
    "            b = info[\"bowler\"]\n",
    "            bowl.setdefault(b, {\"runs\":0, \"wkts\":0})\n",
    "            bowl[b][\"runs\"] += info[\"runs\"][\"total\"]\n",
    "            if \"wicket\" in info:\n",
    "                bowl[b][\"wkts\"] += 1\n",
